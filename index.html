当然可以！下面是一个计算斐波那契数列第n项的Python函数：

def fibonacci(n):
    """计算斐波那契数列的第n项"""
    if n <= 0:
        return "输入必须为正整数"
    elif n == 1:
        return 0
    elif n == 2:
        return 1
    else:
        a, b = 0, 1
        for _ in range(2, n):
            a, b = b, a + b
        return b

# 示例：计算前10项
for i in range(1, 11):
    print(f"斐波那契数列第{i}项: {fibonacci(i)}")
